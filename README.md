# Program menghitung upah gaji per jam seorang pegawai jika perjam 5000


Coding program lengkap

    #include <iostream>
    using namespace std;

    int main()

    {
    int j,hasil;
    cout <<"Program menghitung upah perjam\n";
    cout <<"==============================\n";
    cout <<"Masukkan jumlah jam ="; cin>>j;
    hasil = j*5000;
    cout <<"Jumlah upah keseluruhan =" <<hasil;
    return 0;
    }
